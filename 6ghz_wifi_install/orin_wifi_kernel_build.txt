# This repo includes a pre-built copy of ifx-backports, so you don't need to run this if you don't want to
# Near the end there is some code you need to add to a file. Also, it's just super sketch so I would recommend running it line-by-line and babying the output to make sure you don't fall off the rails <3
# Also, this DOES NOT WORK for Jetpack 6.0, ${WORK_DIR}/Linux_for_Tegra/tools/l4t_update_initrd.sh does not exist while attempting to build the kernel. I assume that it will not work for any other Jetpack version other than 6.2. 

sudo apt install git-core build-essential bc bison flex

WORK_DIR=/home/annika/Documents/LEIDOS/CDA1T/Orin_wifi_build
cd $WORK_DIR
# install packages from https://developer.nvidia.com/embedded/jetson-linux-r3643
# 1. Driver Package (BSP)
# 2. Sample Root Filesystem
# 3. Driver Package (BSP) Sources
# 4. Bootlin Toolchain gcc 11.3 
L4T_GCC_PACKAGE=${WORK_DIR}/aarch64--glibc--stable-2022.08-1.tar.bz2
L4T_RELEASE_PACKAGE=${WORK_DIR}/Jetson_Linux_R36.4.3_aarch64.tbz2
SAMPLE_FS_PACKAGE=${WORK_DIR}/Tegra_Linux_Sample-Root-Filesystem_R36.4.3_aarch64.tbz2
SOURCE_PACKAGE=${WORK_DIR}/public_sources.tbz2

mkdir ${WORK_DIR}/l4t-gcc
cd ${WORK_DIR}/l4t-gcc
tar xf ${L4T_GCC_PACKAGE}

export CROSS_COMPILE=${WORK_DIR}/l4t-gcc/aarch64--glibc--stable-2022.08-1/bin/aarch64-buildroot-linux-gnu-

# Create Initial Image
cd ${WORK_DIR}
tar xf ${L4T_RELEASE_PACKAGE}
sudo tar xpf ${SAMPLE_FS_PACKAGE} -C Linux_for_Tegra/rootfs/
cd Linux_for_Tegra/
sudo ./tools/l4t_flash_prerequisites.sh
sudo ./apply_binaries.sh

# Build Kernel
tar xf ${SOURCE_PACKAGE} -C ${WORK_DIR}/Linux_for_Tegra/..
cd ${WORK_DIR}/Linux_for_Tegra/source
tar xf kernel_src.tbz2
tar xf kernel_oot_modules_src.tbz2
tar xf nvidia_kernel_display_driver_source.tbz2

make -C kernel
export INSTALL_MOD_PATH=${WORK_DIR}/Linux_for_Tegra/rootfs
sudo -E make install -C kernel
cp kernel/kernel-jammy-src/arch/arm64/boot/Image ${WORK_DIR}/Linux_for_Tegra/kernel/Image
export KERNEL_HEADERS=$PWD/kernel/kernel-jammy-src
make modules
sudo -E make modules_install
cd ${WORK_DIR}/Linux_for_Tegra
sudo ./tools/l4t_update_initrd.sh
cd ${WORK_DIR}/Linux_for_Tegra/source
make dtbs
cp kernel-devicetree/generic-dts/dtbs/* ${WORK_DIR}/Linux_for_Tegra/kernel/dtb/

cd $WORK_DIR/Linux_For_Tegra
# Select based on which device you're using
sudo ./flash.sh jetson-orin-nano-devkit internal
# sudo ./flash.sh jetson-agx-orin-devkit internal
# It will take ~10 minutes
# After that, connect a display. If it says "A start job is running for End-user configuration after running OEM installation", reboot it once
# Now go back to your main PC to build the driver



## Now that the kernel is compiled, build the wifi driver off of it
cd $WORK_DIR
git clone git clone https://github.com/Infineon/ifx-backports.git
